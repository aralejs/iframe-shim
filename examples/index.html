遮罩会根据 DOM 的宽高进行调节

.overlay

输入宽高，blur 输入框 

seajs.use(['jquery','../src/iframe-shim'], function($, Shim){
    //example2
    var shim2 = new Shim('#example2 .overlay').sync();

    $('#example2 input').blur(function(e){
        var value = e.currentTarget.value;

        $('#example2 .overlay').css({
            'height': value,
            'width': value
        });

        shim2.sync();
    });
});
